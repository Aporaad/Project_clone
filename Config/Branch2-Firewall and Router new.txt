configure terminal
hostname B2-RE1

! ---------- Domain, NTP ----------
ip domain name hahs.org
clock timezone AST 3
ntp server 192.168.100.10 prefer

! ---------- AAA,USERS,RADIUS ----------
aaa new-model
username arslan privilege 15 secret arslan
radius server NPS_SERVER
 address ipv4 192.168.100.30 auth-port 1812 acct-port 1813
 key 123
 exit
aaa group server radius RADIUS_GROUP
server name NPS_SERVER
exit

aaa authentication login default group RADIUS_GROUP local
aaa authorization exec default group RADIUS_GROUP local
ip radius source-interface Loopback0

line vty 0 4
login authentication default
authorization exec default
transport input all
exit

key chain az
 key 1    
  key-string p2x4pD9R6Y++
! ---------- Interface ----------
interface Loopback0
 ip address 172.16.30.1 255.255.255.255
!
interface GigabitEthernet0/0
 description Link to B2-FG Firewall
 ip address 10.30.13.1 255.255.255.252
 no shutdown
 ip authentication mode eigrp 100 md5
 ip authentication key-chain eigrp 100 az
!
interface GigabitEthernet1/0
 description Link to B2-ASA Firewall
 ip address 10.30.14.1 255.255.255.252
 no shutdown
 ip authentication mode eigrp 100 md5
 ip authentication key-chain eigrp 100 az
!
interface g2/0
 description Link to B2-RE2
 ip address 10.30.12.1 255.255.255.252
 no shutdown
 ip authentication mode eigrp 100 md5
 ip authentication key-chain eigrp 100 az


interface GigabitEthernet3/0
 description Link to ISP 1 for mpls
 ip address 82.1.5.1 255.255.255.252
 no shutdown
 duplex full
!
interface GigabitEthernet4/0
 description Link to ISP 2 for vpn
 ip address 82.2.6.1 255.255.255.252
 no shutdown
!

! ---------- EIGRP ----------
router eigrp 100
 eigrp router-id 172.16.30.1
 network 82.1.5.0 0.0.0.3
 network 10.30.0.0 0.0.255.255 
 network 172.16.30.0 0.0.0.255
 redistribute ospf 1 match internal external 1 external 2 metric 10000 100 255 1 1500

!         
! ---------- OSPF ----------
router ospf 1
 router-id 172.16.30.1
 network 10.30.0.0 0.0.255.255 area 0
 redistribute eigrp 100 subnets

 exit
ip route 0.0.0.0 0.0.0.0 82.1.5.1


! ---------- NAT  ----------
access-list 10 permit  10.30.0.0  0.0.255.255
access-list 10 permit  172.16.30.0 0.0.0.255
ip nat inside source list 10 interface g4/0 overload

int range loopb0,g0/0,g1/0,g2/0
 ip nat inside
int g4/0
 ip nat outside


! ---------- VPN Confige ----------
access-list 100 permit ip 10.30.0.0 0.0.255.255 10.0.0.0 0.255.255.255 
crypto isakmp policy 10
 encr aes 256
 authentication pre-share
 group 5
 lifetime 80000
crypto isakmp key P2X4Phahs address 82.2.1.1
crypto isakmp key P2X4Phahs address 82.2.4.1
crypto ipsec transform-set vpnaz esp-aes esp-sha-hmac
 mode tunnel 
crypto ipsec profile proaz
 set transform-set vpnaz


!------------ TUNNEL ------
interface Tunnel1
 ip address 10.10.20.3 255.255.255.248
 no sh
 no ip redirects
 tunnel source g4/0
 ip nhrp authentication P2X4Phahs
 ip nhrp network-id 123
 tunnel protection ipsec profile proaz
 tunnel mode gre multipoint 

 ip ospf 1 area 0
 ip ospf network point-to-multipoint

 ip nhrp map 10.10.20.1 82.2.1.1
 ip nhrp map multicast 82.2.1.1
 ip nhrp nhs 10.10.20.1
 ip nhrp shortcut
